
        int main()
        {
                int n;
                cin >> n;
                pair<long long, int> s;
                vector<pair<long long, int>> arr;
                for (int i = 0; i < n; i++)
                {
                        cin >> s.first;
                        s.second = 1;
                        arr.push_back(s);
                        cin >> s.first;
                        s.first++;
                        s.second = -1;
                        arr.push_back(s);
                }
                sort(arr.begin(), arr.end());
                long long ans[n + 1];
                for (int i = 0; i < n + 1; i++)
                        ans[i] = 0;
                int t = 0;
                for (int i = 0; i < 2 * n - 1; i++)
                {
                        t += arr[i].second;
                        ans[t] += arr[i + 1].first - arr[i].first;
                }
                for (int i = 1; i < n + 1; i++)
                {
                        cout << ans[i] << " ";
                }
                cout << endl;
                return 0;
        }
        }
