
        int main()
        {
                int n, m;
                cin >> n >> m;
                int r[n];
                int xor_r = 0;
                for (int i = 0; i < n; i++)
                {
                        cin >> r[i];
                        xor_r ^= r[i];
                }
                int c[m];
                int xor_c = 0;
                for (int i = 0; i < m; i++)
                {
                        cin >> c[i];
                        xor_c ^= c[i];
                }
                if (xor_c != xor_r)
                {
                        cout << "NO" << endl;
                        return 0;
                }
                cout << "YES" << endl;
                int final_o = 0;
                for (int i = 0; i < n - 1; i++)
                {
                        int s = 0;
                        for (int j = 0; j < m - 1; j++)
                        {
                                cout << 1 << " ";
                                s ^= 1;
                        }
                        final_o ^= r[i] ^ s;
                        cout << (r[i] ^ s) << endl;
                }
                for (int j = 0; j < m - 1; j++)
                {
                        cout << (((n - 1) % 2) ^ c[j]) << " ";
                }
                cout << (c[m - 1] ^ final_o) << endl;
                return 0;
        }
