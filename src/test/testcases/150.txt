
        int main()
        {
                int n, m;
                cin >> n >> m;
                int arr[n];
                for (int i = 0; i < n; i++)
                        cin >> arr[i];
                int s = 0;
                for (int i = 0; i < n; i++)
                {
                        s += arr[i];
                        cout << s / m << " ";
                        s %= m;
                }
                cout << endl;
                return 0;
        }
