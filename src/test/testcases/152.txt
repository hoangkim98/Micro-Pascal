
        long long x, y, q, n;
        int main()
        {
                cin >> n >> q;
                for (int i = 0; i < q; i++)
                {
                        cin >> y >> x;
                        long long t = 0;
                        t += ((y - 1) / 2) * n;
                        if (x % 2 == y % 2)
                        {
                                if (x % 2 == 0)
                                        t += (n + 1) / 2 + x / 2;
                                else
                                        t += (x + 1) / 2;
                        }
                        else
                        {
                                t += (n * n + 1) / 2;
                                if (x % 2 != 0)
                                        t += n / 2 + (x + 1) / 2;
                                else
                                        t += x / 2;
                        }
                        cout << t << endl;
                }
                return 0;
        }
