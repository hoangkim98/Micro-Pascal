
        int t, n, a[1000010], b[1000010], top;
        int main()
        {
                scanf("%d", &t);
                for (int k = 0; k < t; k++)
                {
                        scanf("%d", &n);
                        for (int j = 0; j < n; j++)
                        {
                                scanf("%d", a + j);
                        }
                        sort(a, a + n);
                        top = 0;
                        for (int i = 1; i < n; i++)
                                if (a[i] == a[i - 1])
                                {
                                        b[top++] = a[i];
                                        a[i] = a[i - 1] = -1;
                                }
                        int s = 1;
                        for (int i = 2; i < top; i++)
                        {
                                if (1ll * b[i - 1] * b[s] > 1ll * b[s - 1] * b[i])
                                        s = i;
                        }
                        cout << b[s] << " " << b[s] << " " << b[s - 1] << " " << b[s - 1] << endl;
                        printf("%d %d %d %d", b[s], b[s], b[s - 1], b[s - 1]);
                }
                return 0;
        }
